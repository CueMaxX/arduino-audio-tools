127 virtual bool begin(int index=0, bool isActive = true) {
128 TRACED();
129 bool result = false;
130 // initilaize volume
131 if (current_volume==-1){
132 setVolume(1.0);
133 } else {
134 setVolume(current_volume);
135 }
136
137 // navigation support
138 autonext = p_source->isAutoNext();
139
140 // start dependent objects
141 p_out_decoding->begin();
142 p_source->begin();
143 meta_out.begin();
144
145 if (index >= 0) {
146 p_input_stream = p_source->selectStream(index);
147 if (p_input_stream != nullptr) {
148 if (meta_active) {
149 copier.setCallbackOnWrite(decodeMetaData, this);
150 }
151 copier.begin(*p_out_decoding, *p_input_stream);
152 timeout = millis() + p_source->timeoutAutoNext();
153 active = isActive;
154 result = true;
155 } else {
156 LOGW("-> begin: no data found");
157 active = false;
158 result = false;
159 }
160 } else {
161 LOGW("-> begin: no stream selected");
162 active = isActive;
163 result = false;
164 }
165 return result;
166 }

384 void moveToNextFileOnTimeout(){
385 if (p_final_stream->availableForWrite()==0) return;
386 if (p_input_stream == nullptr || millis() > timeout) {
387 if (autonext) {
388 LOGW("-> timeout - moving by %d", steam_increment);
389 // open next stream
390 if (!next(steam_increment)) {
391 LOGD("stream is null");
392 }
393 } else {
394 active = false;
395 }
396 timeout = millis() + p_source->timeoutAutoNext();
