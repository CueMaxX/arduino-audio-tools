77 audio_hal_iface_samples_t toSampleRate() {
78 TRACED();
79 const static int ia[] = {8000, 11025, 16000, 22050,
80 24000, 32000, 44100, 48000};
81 const static audio_hal_iface_samples_t oa[] = {
82 AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
83 AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
84 AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
85 int diff = 99999;
86 int result = 0;
87 for (int j = 0; j < 8; j++) {
88 if (ia[j] == sample_rate) {
89 LOGD("-> %d",ia[j])
90 return oa[j];
91 } else {
92 int new_diff = abs(oa[j] - sample_rate);
93 if (new_diff < diff) {
94 result = j;
95 diff = new_diff;
96 }
97 }
98 }
99 LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
100 return oa[result];
101 }

240 virtual size_t readBytes(uint8_t *data, size_t length) override {
241 if (cfg.channels == 2) {
242 return kit.read(data, length);
243 } else if (cfg.channels==1) {
244 // convert 2 channels of int16_t to 1
245 int16_t temp[length];
246 int len_res = kit.read(temp, length*2);
247 int res_count = len_res / 2;
248 int16_t *out = (int16_t*) data;
249 for (int j=0; j<res_count; j+=2){
250 int32_t total = temp[j];
251 total+=temp[j+1];
252 total = total/2;
253 *out = total;
254 out++;
255 }
256 return res_count;
257 }
258 LOGE("Unsuported number of channels : %d", cfg.channels);
259 return 0;

264 virtual void setAudioInfo(AudioBaseInfo info) {
265 TRACEI();
266
267 if (cfg.sample_rate != info.sample_rate
268 && cfg.bits_per_sample == info.bits_per_sample
269 && cfg.channels == info.channels
270 && is_started) {
271 // update sample rate only
272 cfg.sample_rate = info.sample_rate;
273 cfg.logInfo();
274 converter.setAudioInfo(cfg);
275 kit.setSampleRate(cfg.toSampleRate());
276 } else if (cfg.sample_rate != info.sample_rate
277 || cfg.bits_per_sample != info.bits_per_sample
278 || cfg.channels != info.channels
279 || !is_started) {
280 // more has changed and we need to start the processing
281 cfg.sample_rate = info.sample_rate;
282 cfg.bits_per_sample = info.bits_per_sample;
283 cfg.channels = info.channels;
284 cfg.logInfo();
285
286 // Stop first
287 if(is_started){
288 kit.end();
289 }
290 // update input format
291 converter.setAudioInfo(cfg);
292 // start kit with new config
293 kit.begin(cfg.toAudioKitConfig());
294 is_started = true;
295 }

576 AudioActions::ActiveLogic getActionLogic(int pin){
577 input_key_service_info_t input_key_info[] = INPUT_KEY_DEFAULT_INFO();
578 int size = sizeof(input_key_info) / sizeof(input_key_info[0]);
579 for (int j=0; j<size; j++){
580 if (pin == input_key_info[j].act_id){
581 switch(input_key_info[j].type){
582 case PERIPH_ID_ADC_BTN:
583 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveHigh);
584 return AudioActions::ActiveHigh;
585 case PERIPH_ID_BUTTON:
586 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveLow);
587 return AudioActions::ActiveLow;
588 case PERIPH_ID_TOUCH:
589 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveTouch);
590 return AudioActions::ActiveTouch;
591 }
592 }
593 }
594 LOGW("Undefined ActionLogic for pin: %d ",pin);
595 return AudioActions::ActiveLow;

599 void setupActions() {
600 TRACEI();
601 // SPI might have been activated
602 if (!cfg.sd_active){
603 LOGW("Deactivating SPI because SD is not active");
604 SPI.end();
605 }
606
607 // pin conflicts with the SD CS pin for AIThinker and buttons
608 if (! (cfg.sd_active && (AUDIOKIT_BOARD==5 || AUDIOKIT_BOARD==6))){
609 LOGD("actionStartStop")
610 addAction(kit.pinInputMode(), actionStartStop);
611 } else {
612 LOGW("Mode Button ignored because of conflict: %d ",kit.pinInputMode());
613 }
614
615 // pin conflicts with AIThinker A101 and headphone detection
616 if (! (cfg.sd_active && AUDIOKIT_BOARD==6)) {
617 LOGD("actionHeadphoneDetection pin:%d",kit.pinHeadphoneDetect())
618 actions.add(kit.pinHeadphoneDetect(), actionHeadphoneDetection, AudioActions::ActiveChange);
619 } else {
620 LOGW("Headphone detection ignored because of conflict: %d ",kit.pinHeadphoneDetect());
621 }
622
623 // pin conflicts with SD Lyrat SD CS Pin and buttons / Conflict on Audiokit V. 2957
624 if (! (cfg.sd_active && (AUDIOKIT_BOARD==1 || AUDIOKIT_BOARD==7))){
625 LOGD("actionVolumeDown")
626 addAction(kit.pinVolumeDown(), actionVolumeDown);
627 LOGD("actionVolumeUp")
628 addAction(kit.pinVolumeUp(), actionVolumeUp);
629 } else {
630 LOGW("Volume Buttons ignored because of conflict: %d ",kit.pinVolumeDown());
631 }
632 }
