181 TRACED();

187 // find closest frequency for requested sample_rate
188 float freq_requested = cfg.sample_rate * cfg.bits_per_sample ;
189 float selected_freq = 0;
190 for (auto freq : freq_table) {
191 for (auto div : ratio_table) {
192 float freq_value = freq.freq * 1000000 / div.ratio;
193 if (abs(freq_value-freq_requested) < abs(selected_freq-freq_requested)){
194 // MCKFREQ
195 NRF_I2S->CONFIG.MCKFREQ = freq.id << I2S_CONFIG_MCKFREQ_MCKFREQ_Pos;
196 // Ratio
197 NRF_I2S->CONFIG.RATIO = div.id << I2S_CONFIG_RATIO_RATIO_Pos;
198 selected_freq = freq_value;
199 }
200 }
201 LOGI("frequency requested %f vs %f", freq_requested, selected_freq);
