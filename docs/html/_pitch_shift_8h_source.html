155protected:
156 Vector<T> buffer{0};
157 float read_pos_float = 0.0;
158 float cross_fade = 1.0;
159 int write_pos = 0;
160 int write_pointer = 0;
161 int buffer_size = 0;
162 int overlap = 0;
163 float pitch_shift = 0;
164
166 virtual T pitchRead() {
167 TRACED();
168 assert(pitch_shift > 0);
169 assert(buffer_size > 0);
170
171 // read fractional readpointer and generate 0° and 180° read-pointer in
172 // integer
173 int read_pointer_int = roundf(read_pos_float);
174 int read_pointer_int180 = 0;
175 if (read_pointer_int >= buffer_size / 2)
176 read_pointer_int180 = read_pointer_int - (buffer_size / 2);
177 else
178 read_pointer_int180 = read_pointer_int + (buffer_size / 2);
179
180 // read the two samples...
181 float read_sample = (float)buffer[read_pointer_int];
182 float read_sample_180 = (float)buffer[read_pointer_int180];
183
184 // Check if first readpointer starts overlap with write pointer?
185 // if yes -> do cross-fade to second read-pointer
186 if (overlap >= (write_pointer - read_pointer_int) &&
187 (write_pointer - read_pointer_int) >= 0 && pitch_shift != 1.0f) {
188 int rel = write_pointer - read_pointer_int;
189 cross_fade = ((float)rel) / (float)overlap;
190 } else if (write_pointer - read_pointer_int == 0)
191 cross_fade = 0.0f;
192
193 // Check if second readpointer starts overlap with write pointer?
194 // if yes -> do cross-fade to first read-pointer
195 if (overlap >= (write_pointer - read_pointer_int180) &&
196 (write_pointer - read_pointer_int180) >= 0 && pitch_shift != 1.0f) {
197 int rel = write_pointer - read_pointer_int180;
198 cross_fade = 1.0f - ((float)rel) / (float)overlap;
199 } else if (write_pointer - read_pointer_int180 == 0)
200 cross_fade = 1.0f;
201
202 // do cross-fading and sum up
203 T sum = (read_sample * cross_fade + read_sample_180 * (1.0f - cross_fade));
204
205 // increment fractional read-pointer and write-pointer
206 read_pos_float += pitch_shift;
207 if (roundf(read_pos_float) >= buffer_size)
208 read_pos_float = 0.0f;
209
210 return sum;
211 }

221template <typename T> class VariableSpeedRingBuffer : public BaseBuffer<T> {
222public:
223 VariableSpeedRingBuffer(int size = 0, float increment = 1.0) {
224 setIncrement(increment);
225 if (size > 0)
226 resize(size);
227 }
228
229 void setIncrement(float increment) { read_increment = increment; }
230
231 void resize(int size) {
232 buffer_size = size;
233 // prevent an overrun at the start
234 read_pos_float = size / 2;
235 buffer.resize(size);
236 }

238 T read() {
239 assert(read_increment != 0.0);
240 T result = peek();
241 read_pos_float += read_increment;
242 handleReadWriteOverrun(last_value);
243 if (read_pos_float > buffer_size) {
244 read_pos_float -= buffer_size;
245 }
246 return result;
247 }

249 T peek() {
250 if (buffer.size() == 0)
251 return 0;
252 return interpolate(read_pos_float);
253 }

255 bool write(T sample) {
256 if (buffer.size() == 0)
257 return false;
258 handleReadWriteOverrun(last_value);
259 buffer[write_pos++] = sample;
260 // on buffer owerflow reset to 0
261 if (write_pos >= buffer_size) {
262 write_pos = 0;
263 }
264 return true;
265 }
266
268 void reset() {
269 read_pos_float = 0;
270 write_pos = 0;
271 memset(buffer.data(), 0, sizeof(T) * buffer_size);
272 }
273
274 virtual bool isFull() { return false; }
275 virtual int available() { return buffer_size; }
276 virtual int availableForWrite() { return buffer_size; }
277 virtual T *address() { return nullptr; }
278
279protected:
280 Vector<T> buffer{0};
281 int buffer_size;
282 float read_pos_float = 0.0;
283 float read_increment = 0.0;
284 int write_pos = 0;
285 // used to handle overruns:
286 T last_value = 0; // record last read value
287 bool incrementing; // is last read increasing
288
290 T interpolate(float read_pos) {
291 int read_pos_int = read_pos;
292 T value1 = getValue(read_pos_int);
293 T value2 = getValue(read_pos_int + 1);
294 incrementing = value2 - value1 >= 0;
295
296 // make sure that value1 is smaller then value 2
297 if (value2 < value1) {
298 T tmp = value2;
299 value2 = value1;
300 value1 = tmp;
301 }
302 // the result must be between value 1 and value 2: linear interpolation
303 float offset_in = read_pos - read_pos_int; // calculate fraction: e.g 0.5
304 LOGD("read_pos=%f read_pos_int=%d, offset_in=%f", read_pos, read_pos_int, offset_in);
305 float diff_result = abs(value2 - value1); // differrence between values: e.g. 10
306 float offset_result = offset_in * diff_result; // 0.5 * 10 = 5
307 float result = offset_result + value1;
308 LOGD("interpolate %d %d -> %f -> %f", value1, value2, offset_result, result);
309
310 last_value = result;
311
312 return result;
313 }

316 T getValue(int pos) { return buffer[pos % buffer_size]; }

320 bool isMatching(T value1, bool incrementing, T v1, T v2) {
321 bool v_incrementing = v2 - v1 >= 0;
322 // eff sample was ascending so we need to select a ascending value
323 if (incrementing && v_incrementing && value1 >= v1 && value1 <= v2) {
324 return true;
325 }
326 // eff sample was descending so we need to select a descending value
327 if (!incrementing && !v_incrementing && value1 <= v1 && value1 >= v2) {
328 return true;
329 }
330 return false;
331 }
332
335 void handleReadWriteOverrun(T last_value) {
336 // handle overflow - we need to allign the phase
337 int read_pos_int = read_pos_float; // round down
338 if (write_pos == read_pos_int ||
339 write_pos == (buffer_size % (read_pos_int + 1))) {
340 LOGD("handleReadWriteOverrun write_pos=%d read_pos_int=%d", write_pos,
341 read_pos_int);
342 bool found = false;
343
344 // find the closest match for the last value
345 for (int j = read_increment * 2; j < buffer_size; j++) {
346 int pos = read_pos_int + j;
347 float v1 = getValue(pos);
348 float v2 = getValue(pos + 1);
349 // find corresponging matching sample in buffer for last sample
350 if (isMatching(last_value, incrementing, v1, v2)) {
351 // interpolate new position
352 float diff_value = abs(v1 - v2);
353 float diff_last_value = abs(v1 - last_value);
354 float fraction = 0;
355 if (diff_value>0){
356 fraction = diff_last_value / diff_value;
357 }
358
359 read_pos_float = fraction + pos;
360 // move to next value
361 read_pos_float += read_increment;
362 // if we are at the end of the buffer we restart from 0
363 if (read_pos_float > buffer_size) {
364 read_pos_float -= buffer_size;
365 }
366 LOGD("handleReadWriteOverrun -> read_pos pos=%d pos_float=%f", pos, read_pos_float);
367 found = true;
368 break;
369 }
370 }
371 if (!found) {
372 LOGW("phase allign failed: maybe the buffer is too small")
373 }
374 }
375 }

388template <typename T, class BufferT>
389class PitchShiftStream : public AudioPrint {
390public:
391 PitchShiftStream(Print &out) { p_out = &out; }
392
393 PitchShiftInfo defaultConfig() {
394 PitchShiftInfo result;
395 result.bits_per_sample = sizeof(T) * 8;
396 return result;
397 }
398
399 bool begin(PitchShiftInfo info) {
400 TRACED();
401 cfg = info;
402 AudioPrint::setAudioInfo(info);
403 buffer.resize(info.buffer_size);
404 buffer.reset();
405 buffer.setIncrement(info.pitch_shift);
406 active = true;
407 return active;
408 }
409
410 size_t write(const uint8_t *data, size_t len) override {
411 TRACED();
412 if (!active)
413 return 0;
414
415 size_t result = 0;
416 int channels = cfg.channels;
417 T *p_in = (T *)data;
418 int sample_count = len / sizeof(T);
419
420 for (int j = 0; j < sample_count; j += channels) {
421 float value = 0;
422 for (int ch = 0; ch < channels; ch++) {
423 value += p_in[j + ch];
424 }
425 // calculate avg sample value
426 value /= cfg.channels;
427
428 // output values
429 T out_value = pitchShift(value);
430 LOGD("PitchShiftStream %d -> %d", value, out_value);
431 T out_array[channels];
432 for (int ch = 0; ch < channels; ch++) {
433 out_array[ch] = out_value;
434 }
435 result += p_out->write((uint8_t *)out_array, sizeof(T) * channels);
436 }
437 return result;
438 }
439
440 void end() { active = false; }
441
442protected:
443 BufferT buffer;
444 bool active;
445 PitchShiftInfo cfg;
446 Print *p_out = nullptr;
447
448 // execute the pitch shift by writing one sample and returning the pitch
449 // shifted result sample
450 T pitchShift(T value) {
451 TRACED();
452 if (!active)
453 return 0;
454 buffer.write(value);
455 T out_value = buffer.read();
456 return out_value;
457 }
