63class FIR : public Filter<T> {
64 public:
65 template <size_t B>
66 FIR(const T (&b)[B], const T factor=1.0) : lenB(B), factor(factor) {
67 x = new T[lenB]();
68 coeff_b = new T[2*lenB-1];
69 for (uint16_t i = 0; i < 2*lenB-1; i++) {
70 coeff_b[i] = b[(2*lenB - 1 - i)%lenB];
71 }
72 }
73 ~FIR() {
74 delete[] x;
75 delete[] coeff_b;
76 }
77 T process(T value) {
78 x[i_b] = value;
79 T b_terms = 0;
80 T *b_shift = &coeff_b[lenB - i_b - 1];
81 for (uint8_t i = 0; i < lenB; i++) {
82 b_terms += b_shift[i] * x[i] ;
83 }
84 i_b++;
85 if(i_b == lenB)
86 i_b = 0;
87
88#ifdef USE_TYPETRAITS
89 if (!(std::is_same<T, float>::value || std::is_same<T, double>::value)) {
90 b_terms = b_terms / factor;
91 }
92#else
93 if (factor!=1.0) {
94 b_terms = b_terms / factor;
95 }
96#endif
97 return b_terms;
98 }

100 const uint8_t lenB;
101 uint8_t i_b = 0;
102 T *x;

116template <typename T>
117class IIR : public Filter<T> {
118 public:
119 template <size_t B, size_t A>
120 IIR(const T (&b)[B], const T (&_a)[A], T factor=1.0) : factor(factor), lenB(B), lenA(A - 1) {
121 x = new T[lenB]();
122 y = new T[lenA]();
123 coeff_b = new T[2 * lenB - 1];
124 coeff_a = new T[2 * lenA - 1];
125 T a0 = _a[0];
126 const T *a = &_a[1];
127 for (uint16_t i = 0; i < 2 * lenB - 1; i++) {
128 coeff_b[i] = b[(2 * lenB - 1 - i) % lenB] / a0;
129 }
130 for (uint16_t i = 0; i < 2 * lenA - 1; i++) {
131 coeff_a[i] = a[(2 * lenA - 2 - i) % lenA] / a0;
132 }
133 }
134
135 ~IIR() {
136 delete[] x;
137 delete[] y;
138 delete[] coeff_a;
139 delete[] coeff_b;
140 }
141
142 T process(T value) {
143 x[i_b] = value;
144 T b_terms = 0;
145 T *b_shift = &coeff_b[lenB - i_b - 1];
146
147 T a_terms = 0;
148 T *a_shift = &coeff_a[lenA - i_a - 1];
149
150 for (uint8_t i = 0; i < lenB; i++) {
151 b_terms += x[i] * b_shift[i];
152 }
153 for (uint8_t i = 0; i < lenA; i++) {
154 a_terms += y[i] * a_shift[i];
155 }
156
157 T filtered = b_terms - a_terms;
158 y[i_a] = filtered;
159 i_b++;
160 if (i_b == lenB) i_b = 0;
161 i_a++;
162 if (i_a == lenA) i_a = 0;
163
164#ifdef USE_TYPETRAITS
165 if (!(std::is_same<T, float>::value || std::is_same<T, double>::value)) {
166 filtered = filtered / factor;
167 }
168#else
169 if (factor!=1.0) {
170 filtered = filtered / factor;
171 }
172#endif
173 return filtered;
174 }
175
176 private:
177 T factor;
178 const uint8_t lenB, lenA;
179 uint8_t i_b = 0, i_a = 0;
180 T *x;
181 T *y;
182 T *coeff_b;
183 T *coeff_a;
184};
