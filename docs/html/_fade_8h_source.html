77 template <typename T> void convertFrames(T *data, int frames, int channels) {
78 float delta = 1.0 / frames;
79 // handle fade out
80 if (is_fade_in) {
81 fadeIn<T>(data, frames, channels, delta);
82 } else if (is_fade_out) {
83 fadeOut<T>(data, frames, channels, delta);
84 }
85 if (frames > 0) {
86 is_done = true;
87 }

90 template <typename T>
91 void fadeOut(T *data, int frames, int channels, float delta) {
92 for (int j = 0; j < frames; j++) {
93 for (int ch = 0; ch < channels; ch++) {
94 data[j * channels + ch] = data[j * channels + ch] * volume;
95 if (volume > 0) {
96 volume -= delta;
97 if (volume < 0) {
98 volume = 0;
99 }
100 }
101 }
102 }
103 }

105 template <typename T>
106 void fadeIn(T *data, int frames, int channels, float delta) {
107 for (int j = 0; j < frames; j++) {
108 for (int ch = 0; ch < channels; ch++) {
109 data[j * channels + ch] = data[j * channels + ch] * volume;
110 volume += delta;
111 if (volume > 1.0) {
112 volume = 1.0;
113 is_fade_in = false;
114 }
115 }

127class LastSampleFaderT {
128public:
129 void setChannels(int ch) { channels = ch; last.resize(ch); }
130 size_t write(uint8_t *src, size_t size) {
131 if (channels == 0){
132 LOGE("channels=0");
133 return 0;
134 }
135 int frames = size / sizeof(T) / channels;
136 storeLastSamples(frames, src);
137 return size;
138 };
139
142 void end(Print &print, int steps = 200) {
143 T out[channels];
144 for (int j = 0; j < steps; j++) {
145 for (int ch = 0; ch < channels; ch++) {
146 float factor =
147 static_cast<float>(steps - j) / static_cast<float>(steps);
148 out[ch] = last[ch] * factor;
149 }
150 print.write((uint8_t *)out, channels * sizeof(T));
151 }
152 }
153
154protected:
155 int channels = 0;
156 Vector<T> last{0};
157
158 void storeLastSamples(int frames, uint8_t *src) {
159 T *data = (T *)src;
160 for (int ch = 0; ch < channels; ch++) {
161 last[ch] = data[frames - 2 * channels + ch];
162 }
163 }
