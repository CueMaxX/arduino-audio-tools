256 void applyVolume(const uint8_t *buffer, size_t size){
257 switch(info.bits_per_sample){

272 void applyVolume16(int16_t* data, size_t size){
273 for (size_t j=0;j<size;j++){
274 float result = factorForChannel(j%info.channels) * data[j];
275 if (!info.allow_boost){
276 if (result>max_value) result = max_value;
