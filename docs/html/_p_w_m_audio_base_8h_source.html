240 virtual int nextValue() {
241 int result = 0;
242 switch(audio_config.bits_per_sample ){
243 case 8: {
244 int16_t value = buffer->read();
245 if (value<0){
246 LOGE("Could not read full data");
247 value = 0;
248 }
249 result = map(value, -NumberConverter::maxValue(8), NumberConverter::maxValue(8), 0, maxOutputValue());
250 break;
251 }
252 case 16: {
253 int16_t value;
254 if (buffer->readArray((uint8_t*)&value,2)!=2){
255 LOGE("Could not read full data");
256 }
257 result = map(value, -NumberConverter::maxValue(16), NumberConverter::maxValue(16), 0, maxOutputValue());
258 break;
259 }
260 case 24: {
261 int24_t value;
262 if (buffer->readArray((uint8_t*)&value,3)!=3){
263 LOGE("Could not read full data");
264 }
265 result = map((int32_t)value, -NumberConverter::maxValue(24), NumberConverter::maxValue(24), 0, maxOutputValue());
266 break;
267 }
268 case 32: {
269 int32_t value;
270 if (buffer->readArray((uint8_t*)&value,4)!=4){
271 LOGE("Could not read full data");
272 }
273 result = map(value, -NumberConverter::maxValue(32), NumberConverter::maxValue(32), 0, maxOutputValue());
274 break;
275 }
276 }
277 return result;
