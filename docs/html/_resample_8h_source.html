185 size_t write(Print *p_out, const uint8_t* buffer, size_t bytes, size_t &written ) {
186 // prevent npe
187 if (info.channels==0){
188 LOGE("channels is 0");
189 return 0;
190 }
191 T* data = (T*)buffer;
192 int samples = bytes / sizeof(T);
193 size_t frames = samples / info.channels;
194 written = 0;
195
196 // avoid noise if audio does not start with 0
197 if (is_first){
198 is_first = false;
199 setupLastSamples(data, 0);
200 }
201
202 // process all samples
203 while (idx<frames){
204 for (int ch=0;ch<info.channels;ch++){
205 T result = getValue(data, idx, ch);
206 if (p_out->availableForWrite()<sizeof(T)){
207 LOGE("Could not write");
208 }
209 written += p_out->write((uint8_t*)&result, sizeof(T));
210 }
211 idx+=step_size;
212 }
213
214 // save last samples
215 setupLastSamples(data, frames-1);
216 idx -= frames;
217 // returns requested bytes to avoid rewriting of processed bytes
218 return bytes;
219 }

223 // interpolate value
224 int frame_idx1 = frame_idx;
225 int frame_idx0 = frame_idx1-1;
226 T val0 = lookup(data, frame_idx0, channel);
227 T val1 = lookup(data, frame_idx1, channel);

229 float result = mapFloat(frame_idx,frame_idx1,frame_idx0, val0, val1 );

275 T getValue(T *data,float frame_idx, int channel) {
276 // provide value if number w/o digits
277 if (frame_idx==(int)frame_idx) {
278 return ResampleStream<T>::lookup(data, frame_idx-1, channel);
279 }
280 // interpolate value
281 int frame_idx1 = frame_idx;
282 int frame_idx0 = frame_idx1-1;
283 T val0 = ResampleStream<T>::lookup(data, frame_idx0, channel);
284 T val1 = ResampleStream<T>::lookup(data, frame_idx1, channel);
285 T diff = val1 - val0;
286 if (diff==0){
287 // No difference
288 return val0;
289 }
290 // interpolate value
291 float delta = (frame_idx - frame_idx0) - 1;
292 T diffEffective = diff * delta;
293 T result = val0+diffEffective;
294
295 return result;
296 }
