273 // Add samples to input data p_x - and process them if full
274 template<typename T>
275 void processSamples(const void *data, size_t byteCount) {
276 T *dataT = (T*) data;
277 T sample;
278 float sample_windowed;
279 int samples = byteCount/sizeof(T);
280 for (int j=0; j<samples; j+=cfg.channels){
281 sample = dataT[j+cfg.channel_used];
282 sample_windowed = sample;
283 // optionally apply window function
284 if (cfg.window_function!=nullptr){
285 sample_windowed = cfg.window_function->factor(current_pos) * sample;
286 }
287 p_driver->setValue(current_pos, sample_windowed);
288 writeStrideBuffer((uint8_t*)&sample, sizeof(T));
289 if (++current_pos>=cfg.length){
290 fft();
291 }
292 }
293 }

295 void fft() {
296 timestamp_begin = millis();
297 p_driver->fft();
298 timestamp = millis();
299 if (cfg.callback!=nullptr){
300 cfg.callback(*this);
301 }
302
303 // reprocess data in stride buffer
304 if (p_stridebuffer!=nullptr){
305 // rewrite data in stride buffer
306 int byte_count = p_stridebuffer->available();
307 uint8_t buffer[byte_count];
308 p_stridebuffer->readArray(buffer, byte_count);
309 write(buffer, byte_count);
310 current_pos = byte_count / bytesPerSample();
311 } else {
312 current_pos = 0;
313 }
314 }

321 template<int N>
322 bool InsertSorted(AudioFFTResult(&result)[N], AudioFFTResult tmp){
323 for (int j=0;j<N;j++){
324 if (tmp.magnitude>result[j].magnitude){
325 // shift existing values right
326 for (int i=N-2;i>=j;i--){
327 result[i+1] = result[i];
328 }
329 // insert new value
330 result[j]=tmp;
331 }
332 }
333 return false;
334 }

336 void writeStrideBuffer(uint8_t* buffer, size_t len){
337 if (p_stridebuffer!=nullptr){
338 int available = p_stridebuffer->availableForWrite();
339 if (len>available){
340 // clear oldest values to make space
341 int diff = len-available;
342 for(int j=0;j<diff;j++){
343 p_stridebuffer->read();
344 }
345 }
346 p_stridebuffer->writeArray(buffer, len);
347 }
