340 DataNode() = default;
342 DataNode(void*inData, int len){
343 this->len = len;

363 // Assign values from ref, clearing the original ref
364 void assign(DynamicMemoryStream &ref){
365 audio_list.swap(ref.audio_list);
366 it = ref.it;
367 total_available=ref.total_available;
368 default_buffer_size = ref.default_buffer_size;
369 alloc_failed = ref.alloc_failed;;
370 is_loop = ref.is_loop;
371 ref.clear();
372 }
373
375 virtual bool begin() override {
376 clear();
377 temp_audio.resize(default_buffer_size);
378 return true;
379 }
380
381 virtual void end() override {
382 clear();
383 }
384
385 virtual void setLoop(bool loop){
386 is_loop = loop;
387 }
388
389 void clear() {
390 DataNode *p_node;
391 bool ok;
392 do{
393 ok = audio_list.pop_front(p_node);
394 if (ok){
395 delete p_node;
396 }
397 } while (ok);
398
399 temp_audio.reset();
400 total_available = 0;
401 alloc_failed = false;
402 rewind();
403 }
404
405 size_t size(){

407 }

410 void rewind() {
411 it = audio_list.begin();
412 }

414 virtual size_t write(const uint8_t *buffer, size_t size) override {
415 DataNode *p_node = new DataNode((void*)buffer, size);
416 if (p_node->data!=nullptr){
417 alloc_failed = false;
418 total_available += size;
419 audio_list.push_back(p_node);
420
421 // setup interator to point to first record
422 if (it == audio_list.end()){
423 it = audio_list.begin();
424 }
425
426 return size;
427 }
428 alloc_failed = true;
429 return 0;
430 }
431
432 virtual int availableForWrite() override {
433 return alloc_failed ? 0 : default_buffer_size;
434 }
435
436 virtual int available() override {
437 if (it == audio_list.end()){
438 if (is_loop) rewind();
439 if (it == audio_list.end()) {
440 return 0;
441 }
442 }
443 return (*it)->len;
444 }
445
446 virtual size_t readBytes(uint8_t *buffer, size_t length) override {
447 // provide unprocessed data
448 if (temp_audio.available()>0){
449 return temp_audio.readArray(buffer, length);
450 }
451
452 // We have no more data
453 if (it==audio_list.end()){
454 if (is_loop){
455 rewind();
456 } else {
457 // stop the processing
458 return 0;
459 }
460 }
461
462 // provide data from next node
463 DataNode *p_node = *it;
464 int result_len = min(length, (size_t) p_node->len);
465 memcpy(buffer, p_node->data, result_len);
466 // save unprocessed data to temp buffer
467 if (p_node->len>length){
468 uint8_t *start = p_node->data+result_len;
469 int uprocessed_len = p_node->len - length;
470 temp_audio.writeArray(start, uprocessed_len);
471 }
472 //move to next pos
473 ++it;
474 return result_len;
475 }

484 template<typename T>
485 void postProcessSmoothTransition(int channels, float factor = 0.01, int remove=0){
486 if (remove>0){
487 for (int j=0;j<remove;j++){
488 DataNode* node = nullptr;
489 audio_list.pop_front(node);
490 if (node!=nullptr) delete node;
491 node = nullptr;
492 audio_list.pop_back(node);
493 if (node!=nullptr) delete node;
494 }
495 }
496
497 // Remove popping noise
498 SmoothTransition<T> clean_start(channels, true, false, factor);
499 auto first = *list().begin();
500 if (first!=nullptr){
501 clean_start.convert(first->data,first->len);
502 }
503
504 SmoothTransition<T> clean_end(channels, false, true, factor);
505 auto last = * (--(list().end()));
506 if (last!=nullptr){
507 clean_end.convert(last->data,last->len);
508 }
509 }
