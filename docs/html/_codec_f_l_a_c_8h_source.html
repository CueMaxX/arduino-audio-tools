165 static FLAC__StreamDecoderReadStatus read_callback(const FLAC__StreamDecoder *decoder, FLAC__byte result_buffer[],size_t *bytes, void *client_data) {
166 FLAC__StreamDecoderReadStatus result = FLAC__STREAM_DECODER_READ_STATUS_CONTINUE;
167 LOGI("read_callback: %d", *bytes);
168 FLACDecoder *self = (FLACDecoder *)client_data;
169 if (self == nullptr || !self->is_active) {
170 return FLAC__STREAM_DECODER_READ_STATUS_ABORT;
171 }
172
173 // get data directly from stream
174 *bytes = self->readBytes(result_buffer, *bytes);
175 LOGD("-> %d", *bytes);
176 if (self->isEof(*bytes)){
177 result = FLAC__STREAM_DECODER_READ_STATUS_END_OF_STREAM;
178 self->is_active = false;
179 }
180 return result;
181 }

184 bool isEof(int bytes) {
185 bool result = false;
186 if (bytes==0){
187 delay(5);
188 } else {
189 time_last_read=millis();
190 }
191 if (millis() - time_last_read >= read_timeout_ms){
192 result = true;
193 }
194 return result;
195 }

198 static FLAC__StreamDecoderWriteStatus write_callback(const FLAC__StreamDecoder *decoder, const FLAC__Frame *frame,const FLAC__int32 *const buffer[], void *client_data) {
199 LOGI("write_callback: %d", frame->header.blocksize);
200 FLACDecoder *self = (FLACDecoder *)client_data;
201
202 AudioBaseInfo actual_info = self->audioInfo();
203 if (self->info != actual_info){
204 self->info = actual_info;
205 self->info.logInfo();
206 int bps = FLAC__stream_decoder_get_bits_per_sample(decoder);
207 if (bps!=16){
208 LOGI("Converting from %d bits", bps);
209 }
210 if (self->p_notify != nullptr) {
211 self->info = actual_info;
212 self->p_notify->setAudioInfo(self->info);
213 }
214 }
215
216 // write audio data
217 int bps = FLAC__stream_decoder_get_bits_per_sample(decoder);
218 int16_t sample;
219 switch(bps){
220 case 8:
221 for (int j = 0; j < frame->header.blocksize; j++) {
222 for (int i = 0; i < actual_info.channels; i++) {
223 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j]<<8;
224 sample = buffer[i][j]<<8;;
225 self->p_print->write((uint8_t *)&sample,2);
226 }
227 }
228 break;
229 case 16:
230 for (int j = 0; j < frame->header.blocksize; j++) {
231 for (int i = 0; i < actual_info.channels; i++) {
232 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j];
233 sample = buffer[i][j];
234 self->p_print->write((uint8_t *)&sample,2);
235 }
236 }
237 break;
238 case 24:
239 for (int j = 0; j < frame->header.blocksize; j++) {
240 for (int i = 0; i < actual_info.channels; i++) {
241 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j]>>8;
242 sample = buffer[i][j] >>8;
243 self->p_print->write((uint8_t *)&sample,2);
244 }
245 }
246 break;
247 case 32:
248 for (int j = 0; j < frame->header.blocksize; j++) {
249 for (int i = 0; i < actual_info.channels; i++) {
250 //self->output_buffer[j*actual_info.channels+ i] = buffer[i][j]>>16;
251 sample = buffer[i][j] >>16;
252 self->p_print->write((uint8_t *)&sample,2);
253 }
254 }
255 break;
256 default:
257 LOGE("Unsupported bps: %d", bps);
258 }
259
260 return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;
261 }

363 virtual size_t write(const void *in_ptr, size_t in_size) override {
364 if (!is_open || p_print == nullptr) return 0;
365 LOGD("write: %u", in_size);
366 size_t result = 0;
367 int samples=0;
368 int frames=0;
369 int32_t *data=nullptr;
370 switch(cfg.bits_per_sample){
371 case 16:
372 samples = in_size / sizeof(int16_t);
373 frames = samples / cfg.channels;
374 writeBuffer((int16_t*)in_ptr, samples);
375 data = buffer.data();
376 break;
377
378 case 32:
379 samples = in_size / sizeof(int32_t);
380 frames = samples / cfg.channels;
381 data = (int32_t*) in_ptr;
382 break;
383
384 default:
385 LOGE("bits_per_sample not supported: %d", cfg.bits_per_sample);
386 break;
387 }
388
389 if (frames>0){
390 if (FLAC__stream_encoder_process_interleaved(p_encoder, data, frames)){
391 result = in_size;
392 } else {
393 LOGE("FLAC__stream_encoder_process_interleaved");
394 }
395 }
