145class AudioSourceURL : public AudioSource {
146public:
147 template<typename T, size_t N>
148 AudioSourceURL(AbstractURLStream& urlStream, T(&urlArray)[N], const char* mime, int startPos = 0) {
149 TRACED();
150 this->actual_stream = &urlStream;
151 this->mime = mime;
152 this->urlArray = urlArray;
153 this->max = N;
154 this->pos = startPos - 1;
155 this->timeout_auto_next_value = 20000;
156 }
157
159 virtual void begin() override {
160 TRACED();
161 this->pos = 0;
162 }
163
165 Stream* selectStream(int idx) override {
166 pos = idx;
167 if (pos < 0) {
168 pos = 0;
169 LOGI("url array out of limits: %d -> %d", idx, pos);
170 }
171 if (pos >= max) {
172 pos = max-1;
173 LOGI("url array out of limits: %d -> %d", idx, pos);
174 }
175 LOGI("selectStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
176 if (started) actual_stream->end();
177 actual_stream->begin(urlArray[pos], mime);
178 started = true;
179 return actual_stream;
180 }
181
183 Stream* nextStream(int offset) override {
184 pos += offset;
185 if (pos < 0 || pos >= max) {
186 pos = 0;
187 }
188 LOGI("nextStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
189 return selectStream(pos);
190 }
191
193 Stream* previousStream(int offset) override {
194 pos -= offset;
195 if (pos < 0 || pos >= max) {
196 pos = max-1;
197 }
198 LOGI("previousStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
199 return selectStream(pos);
200 }
201
203 Stream* selectStream(const char* path) override {
204 LOGI("selectStream: %s", path);
205 if (started) actual_stream->end();
206 actual_stream->begin(path, mime);
207 started = true;
208 return actual_stream;
209 }
210
211 int index() {
212 return pos;
213 }
214
215 const char *toStr() {
216 return urlArray[pos];
217 }
218
220 void setTimeout(int millisec){
221 actual_stream->setTimeout(millisec);
222 }
223
224 // provides go not to the next on error
225 virtual bool isAutoNext() {
226 return true;
227 };
228

231 TRACEI();

233 }
234
235
236protected:
