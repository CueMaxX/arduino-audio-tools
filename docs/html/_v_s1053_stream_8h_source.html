101 bool begin(VS1053Config cfg) {
102 TRACEI();
103 bool result = true;
104 // enfornce encoded data for midi mode
105 if (cfg.is_midi){
106 cfg.is_encoded_data = true;
107 }
108 this->cfg = cfg;
109 setAudioInfo(cfg);
110 cfg.logInfo();
111 LOGI("is_encoded_data: %s", cfg.is_encoded_data?"true":"false");
112 LOGI("is_midi: %s", cfg.is_midi?"true":"false");
113 LOGI("cs_pin: %d", cfg.cs_pin);
114 LOGI("dcs_pin: %d", cfg.dcs_pin);
115 LOGI("dreq_pin: %d", cfg.dreq_pin);
116 LOGI("reset_pin: %d", cfg.reset_pin);
117 LOGI("cs_sd_pin: %d", cfg.cs_sd_pin);
118
119 if (p_vs1053==nullptr){
120 p_vs1053 = new VS1053(cfg.cs_pin,cfg.dcs_pin,cfg.dreq_pin);
121 p_vs1053_out = new VS1053StreamOut(p_vs1053);
122
123 if (cfg.is_start_spi) {
124 LOGI("SPI.begin()")
125 SPI.begin();
126 } else {
127 LOGI("SPI not started");
128 }
129
130 if (cfg.reset_pin!=-1){
131 LOGI("Setting reset pin to high: %d", cfg.reset_pin);
132 pinMode(cfg.reset_pin, OUTPUT);
133 digitalWrite(cfg.reset_pin, HIGH);
134 delay(800);
135 }
136 }
137
138 // Output Stream
139 if (p_out==nullptr){
140 AudioEncoder *p_enc = cfg.is_encoded_data ? &copy:p_encoder;
141 p_out = new EncodedAudioStream(p_vs1053_out, p_enc);
142 }
143
144 // hack to treat midi as separate mode
145 const int MIDI_MODE = 100;
146 int mode = cfg.mode;
147 if (cfg.is_midi){
148 mode = MIDI_MODE;
149 }
150
151 switch(mode){
152
153 case TX_MODE:
154 result = beginTx();
155 break;
156#if VS1053_EXT
157 case MIDI_MODE:
158 result = beginMidi();
159 break;
160
161 case RX_MODE:
162 result = beginRx();
163 break;
164#endif
165 default:
166 LOGD("Mode not supported");
167 result = false;
168 break;
169 }
170 return result;
171 }
